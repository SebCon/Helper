 35   /** check if value is integer
 36   *		@function isInteger
 37   *		@param {(string|number)} value as number or as string
 38   *
 39   *   @return {boolean} is value a number
 40   **/
 41   var isInteger = function(value) {
 42     var back = false;
 43     if (value !== undefined && value !== null) {
 44       var patt = /^\+?[0-9]+$/;
 45       back = patt.test(value);
 46       //var val = parseInt(value);
 47       //back = (!isNaN(val));
 48     }
 49     return back;
 50   };

 75   /** first letter will transform to capital letter
 76   *		@function changeFirstLetterToCapital
 77   *		@param {string} text any text
 78   *
 79   *   @return {string} transform text
 80   **/
 81   var changeFirstLetterToCapital = function(text) {
 82     var result = "";
 83     if (text) {
 84       var arr = text.split(" ");
 85       var l = arr.length;
 86       for (var i = 0; i < l; i++) {
 87         var t = arr[i];
 88         t = t.charAt(0).toUpperCase() + t.slice(1);
 89         result = result + t ;
 90         // Dont add space to last part
 91         if (i < (arr.length - 1)) {
 92           result = result + " ";
 93         }
 94       }
 95     }
 96     return result;
 97   };

100   /** create random id via user/standard config
101   *
102   *   @generator
103   *		@function createRandomId
104   *		@param {Object} config config object
105   *   @param {number} [config.len = 5] length of random id
106   *   @param {string} [config.chars = aA#] random id mask
107   *
108   *   @return {string} random id as string
109   **/
110   var createRandomId = function(config) {
111     var result = '';
112 
113     if (config) {
114       var length = config.len || 5;
115       var chars = config.chars || 'aA#';
116       var mask = '';
117 
118       if (chars.indexOf('a') > -1) mask += 'abcdefghijklmnopqrstuvwxyz';
119       if (chars.indexOf('A') > -1) mask += 'ABCDEFGHIJKLMNOPQRSTUVWXYZ';
120       if (chars.indexOf('#') > -1) mask += '0123456789';
121       if (chars.indexOf('!') > -1) mask += '~`!@#$%^&*()_+-={}[]:";\'<>?,./|\\';
122       for (var i = length; i > 0; --i) {
123         result += mask[Math.floor(Math.random() * mask.length)];
124       }
125     }
126 
127     return result;
128   };

131   /** remove all input/textarea values of parent element
132   *		@function removeAllInputContent
133   *		@param {string} id parent element id
134   **/
135   var removeAllInputContent = function(id) {
136     if (id) {
137       var parent = document.getElementById(id);
138       if (parent) {
139         var childsInput = parent.getElementsByTagName('input');
140         var childsTextArea = parent.getElementsByTagName('textarea');
141 
142         if (childsInput) {
143           for (var i=0; i < childsInput.length; i++) {
144             var elem = childsInput[i];
145             if (elem) {
146               elem.value = '';
147             }
148           }
149         }
150 
151         if (childsTextArea) {
152           for (var k=0; k < childsTextArea.length; k++) {
153             var elem2 = childsTextArea[k];
154             if (elem2) {
155               elem2.innerHTML = '';
156             }
157           }
158         }
159 
160       }
161     }
162   };

178   /** start interval with interval id
179   *		@function startInterval
180   *		@param {string} id id of interval
181   *   @param {function} callback callback function
182   *   @param {number} [time = 250] delay time
183   **/
184   var startInterval = function(id, callback, time) {
185     if (id && time && callback && typeof callback === 'function') {
186       console.log('start interval');
187       if (intervals[id]) {
188         stopInterval(id);
189       }
190       time = isInteger(time) ? time : 250;
191       intervals[id] = setInterval(callback, time);
192     } else {
193       console.warn('cannot set interval!');
194     }
195   };

198   /** transform text within char limit
199   *		@function getLimitString
200   *		@param {string} value text
201   *   @param {number} count max char count
202   *   @param (boolean) [showDots] show dots if text length is greater than char count
203   *
204   *   @return {string} transformed text
205   **/
206   var getLimitString = function(value, count, showDots) {
207     var lStr = value;
208     if (value && count) {
209       lStr = value.substring(0, count);
210       if (showDots) {
211         if ((count + 4) >= value.length) {
212           lStr = value;
213         } else if (value.length > count) {
214           lStr += ' ...';
215         }
216       }
217     }
218 
219     return lStr;
220   };

223   /** convert unix time stamp into readable date
224   *		@function convertUnixTimestampToDate
225   *		@param {number} tstamp unix timestamp
226   *   @param {object} [options] config object
227   *   @param {string} [options.year = numeric] year format
228   *   @param {string} [options.month = numeric] month format
229   *   @param {string} [options.day = numeric] day format
230   *   @param {string} [options.hour = numeric] hour format
231   *   @param {string} [options.minute = numeric] minute format
232   *   @param {string} [options.second = numeric] second format
233   *
234   *   @return {string} convert timestamp
235   **/
236   var convertUnixTimestampToDate = function(tstamp, options) {
237     var formattedTime = null;
238     if (tstamp !== undefined && tstamp !== null) {
239       // author: https://stackoverflow.com/questions/847185/convert-a-unix-timestamp-to-time-in-javascript
240       // Create a new JavaScript Date object based on the timestamp
241       // multiplied by 1000 so that the argument is in milliseconds, not seconds.
242       var date = new Date(tstamp * 1000);
243       if (!options) {
244         options = { year: 'numeric', month: 'numeric', day: 'numeric', hour: 'numeric', minute: 'numeric', second: 'numeric' };
245       }
246       // @todo change german lanuage
247       formattedTime = date.toLocaleDateString('de-DE', options);
248     }
249 
250     return formattedTime;
251   };

257   *   @param {*} value value to save
258   **/
259   var setHash = function(id, value) {

263       }

265     }
